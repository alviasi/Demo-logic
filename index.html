async function loadFromDrive() { const url = "https://docs.google.com/spreadsheets/d/1IWFngGmDQ40FOA1NTeaNoUNY7hcfOevf/export?format=xlsx"; const response = await fetch(url); const data = await response.arrayBuffer(); const workbook = XLSX.read(data, { type: "array" }); const sheetName = workbook.SheetNames[0]; const worksheet = workbook.Sheets[sheetName]; const cell = worksheet["C7"]; const value = cell ? cell.v : ""; document.getElementById("box4").value = value; }
